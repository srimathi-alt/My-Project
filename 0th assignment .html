# Function to categorize columns
import pandas as pd

# Example: manually define which columns are ordinal (based on domain knowledge)
ordinal_columns = ['Satisfaction Level', 'Education Level']  # Replace or add your own

def categorize_columns(df, ordinal_cols=None):
    if ordinal_cols is None:
        ordinal_cols = []

    qualitative_nominal = []
    qualitative_ordinal = []
    quantitative_discrete = []
    quantitative_continuous = []

    for column in df.columns:
        if column in ordinal_cols:
            qualitative_ordinal.append(column)
        elif df[column].dtype == 'object':
            qualitative_nominal.append(column)
        elif df[column].dtype in ['int64', 'float64']:
            unique_vals = df[column].nunique()
            if df[column].dtype == 'int64' and unique_vals < 50:
                quantitative_discrete.append(column)
            else:
                quantitative_continuous.append(column)

    return {
        "Qualitative - Nominal": qualitative_nominal,
        "Qualitative - Ordinal": qualitative_ordinal,
        "Quantitative - Discrete": quantitative_discrete,
        "Quantitative - Continuous": quantitative_continuous
    }

# Call the function with the DataFrame and ordinal column names
column_categories = categorize_columns(df, ordinal_cols=['Senior Citizen'])  # example: treating "Senior Citizen" as ordinal

# Convert the result into a structured DataFrame
category_data = []
for category, columns in column_categories.items():
    for column in columns:
        category_data.append({
            "Column Name": column,
            "Type": category.split(" - ")[0],
            "Sub-category": category.split(" - ")[1]
        })

category_df = pd.DataFrame(category_data)

# Display the final categorization
print(category_df)

def descriptive_stats(df):
    numeric_df = df.select_dtypes(include=['int64', 'float64'])

    results = []

    for col in numeric_df.columns:
        col_data = numeric_df[col]

        stats = {
            "Column": col,
            "Mean": col_data.mean(),
            "Median": col_data.median(),
            "Mode": col_data.mode().iloc[0] if not col_data.mode().empty else None,
            "Range": col_data.max() - col_data.min(),
            "Variance": col_data.var(),
            "Std Deviation": col_data.std(),
            "Min": col_data.min(),
            "Max": col_data.max(),
            "IQR (Spread)": col_data.quantile(0.75) - col_data.quantile(0.25)
        }

        results.append(stats)
    return pd.DataFrame(results)

# Run the function
stats_df = descriptive_stats(df)

# Display the final stats
print(stats_df)
